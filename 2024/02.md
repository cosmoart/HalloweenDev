function battleHorde (zombies, humans) {
	let zTotal = 0, hTotal = 0

	for (const zombie of zombies) {
		zTotal += +zombie
	}
	for (const human of humans) {
		hTotal += +human
	}

  return zTotal === hTotal ? "x" : zTotal > hTotal
  ? zTotal - hTotal + "z" : hTotal - zTotal + "h";
